# let和const

es6对于变量方面新增了let和const，相较于之前的var有很大的区别。

变量名称 var let const
作用域 有全局作用域 只能限制于作用域内 只能限制于作用域内
变量名提升 存在 不存在 不存在
是否可修改赋值 可以重新赋值 可以 不能
能否多次声明 可以 不能 不能

   //reduce方法，reudce(pre,cur,index,arr)

# 扩展运算符（三个点）

TIP

总结下就是将数组展开，或者将数组收起，以及接收多余的值

基本用法

let arr = [1,2,3,5,0];
    console.log(...arr);  //可以将数组展开 1 2 3 5 0


function f(...a) {  //不仅可以将数组展开，也能将数组收起
     return a.sort() //快速排序的方式
    }

 console.log(f(3, 2, 1));   //Array(3) [ 1, 2, 3 ]


//还能将剩下没取到的给接受
    function g(a,b,...c){
        console.log(a, b, c); // 1 2 Array(4) [ 3, 4, 5, 6 ]
    }
    g(1,2,3,4,5,6);

便利用法，将数组里的值快速取出并当函数的实参传入。

function p(a,b,c){

    }
    p(...[2,3,5])

便利用法，快速的将数组复制一份（复制后不相等）

    //将原始数据映射一次
    let arr2 = [1,2,5,9,5];
    //如果想要复制一份
    let arr3 =[...arr2];

    console.log(arr2, arr3); //Array(5) [ 1, 2, 5, 9, 5 ]   Array(5) [ 1, 2, 5, 9, 5 ]
    console.log(arr2 == arr3);  //false

# 解构赋值

对象的解构，可以省去用之前点语法的方式将属性值求出来的麻烦

    let json = {
        name:'zhangsan',
        age:20,
        job:'jobs'
    }

    //可用于解构json
    let {name,age,job} = json;

//解构的时候还可以起别的名字 name:n
    let {name:h,age:j,job:k} = json;
    console.log(h, j, k);

数组的解构，快速将里面的值取出

   let arr = [1,2,3,4]
    let [t,y,u,i] = arr;   //左右的结构要一样,如果不一样则返回undefined
    console.log(t, y, u, i); // 1 2 3 4

如果没有给对应的数据，可以给个默认值
	let [aa,cc,rr,tt,ll=7] = arr;
    console.log(aa, cc, rr, tt,ll); // 1 2 3 4 7

//利用解构，交换两个值的位置

    let x = 5;
    let v = 15;
    [x,v] = [v,x];
    console.log(x, v); //15 5

对象的复制：

   //复制对象,不会影响之前的
    let arr = [1,2,3];
    let arr2 = Object.assign([],arr);
    arr2.push(4);
    console.log(arr2);  // Array(4) [ 1, 2, 3, 4 ]

对象的遍历

新增的遍历方法，for...of 和es5的for.. key in value 差不多 但是会更加的灵活，使用Object.keys(.....),Object.values(.....) 以及Object.entries(.....)可以分别循环出，键，值，实体

   //Object.keys();
    let json = {
        a:1,
        b:2,
        c:3
    };
    for(let key of Object.keys(json)){   // 和for.. key in value 差不多
        console.log(key);   // 打印出键值 a,b,c
    };

    //Object.values()
    for(let value of Object.values(json)){
        console.log(value);  // 打印出值 1，2，3
    }
    //循环实体 Object.entries
    for(let item of Object.entries(json)){
        console.log(item); // Array [ "a", 1 ]  .......
    }

# 新增数据结构set

es6推出了类似数组的set这个新的数据结构。特点是不能重复

基本使用

	let arr = ['a','b','c','a'];
    let set = new Set(['a','b','c','a']);
    console.log(arr);
    console.log(set);   //重复的a没有被打印出来

set也有类似数组的增删改以及循环的方法

    let set1 = new Set();
    set1.add('a');
    set1.add('b');
    set1.add('c');
    set1.add('a');   // 重复的不会被添加
    set1.delete('b');   //会将对应的值删掉

    console.log(set1.has('c'));   //会判断里面有没有            true
    console.log(set1.size);    //类似数组的length属性，显示个数  2
    '----------------------------------'
    //clear方法清除所有
    set1.clear()
    console.log(set1);   


    //遍历set数据结构
    let set3 = new Set(['a','v','s']);
    for(let item of set3.values()){    //如果不加value()这里还是默认按value来遍历
        console.log(item); // a,v,s
    }

    for(let item of set3.keys()){
        console.log(item);  // a,v,s
    }

    for(let item of set3.entries()){
        console.log(item);           //打印实体 Array [ "a", "a" ]  Array [ "v", "v" ]  Array [ "s", "s" ]
    }


不过循环的话，还是用foreach会方便一点

    set3.forEach((v,i)=>{
        console.log(v);
        console.log(i);  //打印的值和v一样
    })

# set数据结构用法——数组去重

 	let arr = [12,3,45,6,3,12];
    let set = new Set(arr);
    console.log(set);  //重复的不会显示  Set(4) [ 12, 3, 45, 6 ]

WARNING

但是类型还是set类型啊，还是不如操作数据的方法多。所以就需要对其进行转换，使用扩展运算符的方式,使用...可以直接转换

# generator函数

TIP

处理异步的函数，可以理解为是一个过渡的方法（过渡后为async和await） 过于资料的描述： Generator函数是将函数分步骤阻塞 ，只有主动调用next() 才能进行下一步 ，因为asyns函数相当于Generator函数的语法糖，做出了优化，所以这里对Generator函数不做赘述，而且一般用到异步编程的时候一般也只用async和promise。所以这里就省略了,但是因为dva中异步处理的实现就是Generator函数，所以还是挺有用的（但是现在可能用的少），当然dva中自己带状态机使函数自动运行下去。

基本用法

    //generator函数用法,声明时带*，内部配合yield。最后通过next()进行调用
    function * g(){
        yield 'generator函数第一个'
        yield 'generator函数第二个'
        yield 'generator函数第三个'
        return '函数调用完毕'
    };

    let gen = g();
    //里面写了几个就调用几次
    console.log(gen.next());  // Object { value: "generator函数第一个", done: false }
    console.log(gen.next());   //Object { value: "generator函数第二个", done: false }
    console.log(gen.next());   //{ value: "generator函数第三个", done: false }
    console.log(gen.next());  //{ value: "函数调用完毕", done: true }
    //最后调用完毕后，会返回done=true




    //可以用for...of循环去自动遍历。。但是通过for..of去循环的，无法将return的内容循环出
    for(let item of gen){
        console.log(item);
        //generator函数第一个
        //generator函数第二个
        //generator函数第三个
    }



    //同样还可以通过解构来解构出来。同样的无法解构出return的内容。因为那个是函数的返回值
    let [a,b,c,d] = g();
    console.log(a, b, c, d); // generator函数第一个 generator函数第二个 generator函数第三个 undefined

    let[x,...y]=g();
    console.log(x, y);   //还可以通过扩展运算符将剩余的并入y中成为数组generator函数第一个    Array [ "generator函数第二个", "generator函数第三个" ]



    //或者直接通过扩展运算符解出来

    console.log(...g());  //generator函数第一个 generator函数第二个 generator函数第三个

    //也可以通过Array.from() 直接转出为数组
    console.log(Array.from(g()));  //Array(3) [ "generator函数第一个", "generator函数第二个", "generator函数第三个" ]

TIP

那么说来说去，有什么好处？之前可能会遇到ajax的疯狂嵌套问题，嵌套多层，容易混淆，但是现在这个方法用起来就是解决这个问题

一下为模版示例

    function * getApi(){
        let val = yield '?tc_name=前端&tc_avatar=123456';
        yield axios.post(`http://.....}`)
    }

    let a = getApi();
    let username = a.next().value
    a.next(username).value.then((res)=>{
        console.log(res);
    })

# Promise 函数

promise比较简单，也是最常用的，主要就是将原来的用回调函数的异步编程方法转成用relsove和reject触发事件， 用then和catch捕获成功或者失败的状态执行相应代码的异步编程的方法

以下用promise模拟登录

  
   let status = 1;   //模拟成功的状态值

   //去登录
   let log = (resolve,reject)=>{
       setTimeout(()=>{
           if(status==1){
               resolve('登录成功，返回信息')
           }else {
               reject('登录失败，回去吧')
           }
       },2000);
   };

   //登录后获取用户信息
   let getInfo = (resolve,reject)=>{
       setTimeout(()=>{
           if(status==1){
               resolve('信息获取成功');
           }else{
               reject('信息获取失败')
           }
       },2000);
   };

   //以上是登录逻辑，下面开始实施
   new Promise(log).then((res)=>{
       console.log(res);
       return new Promise(getInfo);   //登录成功，向后台获取用户信息.直接将登录成功的信息return到后台
   }).then((res)=>{  //接收到后台返回的用户信息，然后显示
       console.log(res);
   })

由上面代码可以看出 promise将多个回调函数嵌套的回调地狱 ，变成了链式的写法 ，可读性更高写法也更清晰

# async函数

简单的说async函数就相当于自执行的Generator函数，相当于自带一个状态机，在await的部分等待返回， 返回后自动执行下一步。而且相较于Promise,async的优越性就是把每次异步返回的结果从then中拿到最外层的方法中，不需要链式调用，只要用同步的写法就可以了。更加直观而且，更适合处理并发调用的问题。但是async返回的本质上是一个promise函数。所以必须以一个Promise对象开始 ，所以async通常是和Promise结合使用的

 async function f() {
        //如果await后面跟了reject，那么就直接出现问题了.后面的代码也不走了

        await Promise.reject('出现问题了')
        let a1 = await Promise.resolve('没毛病')
        console.log(a1);

    }


    f().then(res=>{
        console.log(res);
    }).catch(err=>{
        console.log(err);  //打印错误
    })


    //如何解决async函数抛出的错误 使用try..catch
    async function t(){
        try{
            await Promise.reject('报错了');
        }catch(e){
            // console.log(e);   //报错信息会在这打印出来，但是不会阻碍后续代码的执行
        }

        let a = await Promise.resolve('又好了')
        console.log(a);
    }

    t().then(res=>{
        console.log(res);
    }).catch(err=>{
        console.log(err);
    })


    //或者使用promise本身的catch去解决

    async function p(){
        await Promise.reject('使用本身的去解决').catch(err=>{
            console.log(err);
        })
        let b = await Promise.resolve('用promise本身方法解决')
        console.log(b);


    }
    p().then(res=>{
        console.log(res);
    })

# 扩展类的方法

以往给原型上加属性通过prototype，或者用assign

	function Person(name,age){
        this.name=name;
        this.age=age;
    };
    Person.prototype.show = function(){
        return `这是show里面的方法，名字是${this.name}`
    }
    let p1 = new Person('张三',19);
    console.log(p1.show());


	//还可以用es6的扩展对象方式添加
    Object.assign(Person.prototype,{
        speak(){
            return `这是speak里的方法，年龄为${this.age}`;
        }
    })
    console.log(p1.speak());
