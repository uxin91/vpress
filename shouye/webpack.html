# webpack 1.0 配置

webpack配置比较繁琐，有几次尝试自己去配，但是后面一段时间不用就又忘了，蛋疼，又得翻那坑货文档。保存了的1.0的配置，常见情况下应该不会出问题。比较节约时间。

var htmlWp = require('html-webpack-plugin')  //html解析
var path = require('path')                   //路径
module.exports = {
    entry:'./src/main.js',
    output:{
        path:__dirname+'/dist',
        filename:'build.js'
    }
    ,
    plugins:[
        new htmlWp({
            title: 'shouye',
            template: './src/index.html'
        })
    ],
    module:{
        loaders:[
            {
                test:/\.js$/,
                loader:'babel-loader',
                exclude:/node_modules/,
            },
            //webpack对es6的语法不够会报错，在下面单独为muse-ui加上需要解析的规则
            {
                test:/muse-ui.dist.*?js$/,
                loader:'babel-loader'
            },
            {
                test:/\.vue$/,
                loader:'vue-loader'
            },
            {
                test:/\.css/,
                loader:'style-loader!css-loader'
            },

            {
                test: /\.(jpg|png|gif|ttf|svg|woff|svg)$/,
                loader: "url-loader?limit=40000"
            },
            {
                test: /\.js$/,  // 将.js文件中的es6语法转成es5语法
                loader:'babel-loader',
                exclude:/node_modules/
            },
            {
                test: /vue-preview.src.*?js$/,  // vue-preivew组件专用
                loader: 'babel'
            }
        ]
    },
    babel:{
        presets:['es2015'],
        plugins:['transform-runtime']
    }
}

WARNING

综上，可以看出有的ui框架没法很好的解析，这里建议还是再重新配置个规则
