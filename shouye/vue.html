# 脚手架安装

  1. npm install -g @vue/cli
  2. npm install -g @vue/cli-service-global
  3. vue create hello-world
  4. npm run dev 运行
  5. npm run build 编译

# 通过url的参数传递

  • 路由设置，和react原理一样 {path:'/list/newInfo/:id',component:info} 通过:id

  • 接收页面设置 this.$route.params.id 注意前后值一样，例如传来的是id这里就用id接收

# 父组件向子组件传值

  • 父组件在引用的子组件上挂载：<commen :id='id'></commen>
  • 子组件通过props:[id] 接收 props:['id']

# 子组件向父组件传值

父组件设置

  1. 引入子组件 import fater from "./components/fater";
  2. 使用并挂载方法：<fater @getChild='getChild'></fater>
  3. 父组件中定义方法，作为接收：
	getChild(mes){
  		console.log(mes);
  		this.fats = mes;
	}

WARNING

需要注意的是选择的挂载方法要都是父组件的方法（就是提醒父组件这监听方法的时候别搞错了）

<fater @getChild='getChild'>

完整代码：

父组件

<template>
  <div id="app">
    父组件
    <hr>
    <fater @getChild='getChild'></fater>
    <span>{{fats}}</span>
  </div>
</template>

<script>
import fater from "./components/fater";

export default {
  name: 'App',
  components: {
    fater
  },
 data(){
    return{
      fats:''
    }
 },
  methods:{
    getChild(mes){
      console.log(mes);
      this.fats=mes;
    }
  }
}
</script>

<style>

</style>

子组件：

<template>
    <div>
        子组件
        <button @click="send">传递消息给父组件</button>
    </div>
</template>
<script>
    export default {
        methods:{
            send(){
                this.$emit('getChild','子组件传递而来')
            }
        }
    }
</script>

# 父组件获取子组件的方法

先在子组件里定义好一个方法

			sons(){
                console.log('子组件的方法');
            }

父组件中设置：

  • ref='子组件名'进行监听 <fater ref='fater'></fater>
  • 定义一个方法来触发，并通过this.$refs.fater.sons()来获取
	getsons(){
      console.log(this.$refs.fater.sons());
    }

完整代码 子组件中设置：

<template>
    <div>
        子组件
    </div>
</template>
<script>
    export default {
        methods:{   
            sons(){
                console.log('子组件的方法');
            }
        }
    }
</script>
<style>

</style>

父组件中设置：

<template>
  <div id="app">
    父组件
    <button @click='getsons'>获取子组件的方法</button>
    <hr>
    <fater ref='fater'></fater>
  </div>
</template>

<script>
import fater from "./components/fater";

export default {
  name: 'App',
  components: {
    fater
  },
 data(){
    return{
      
    }
 },
  methods:{
    getsons(){
      console.log(this.$refs.fater.sons());
    }
  }
}
</script>

# 子组件获取父组件的方法

父组件设置：

  1. 先在父组件中定义一个方法
faterMethod(){
      console.log('父组件中的方法');
    }
  1. 然后在引用的子组件上挂载 <fater @faterMethod='faterMethod'></fater>

子组件设置：

包装在方法中，通过

			getFaterMeth(){
                this.$emit('faterMethod','hello')
            }

获取

完整代码：

父组件：

<template>
  <div id="app">
    父组件
    <hr>
    <fater @faterMethod='faterMethod'></fater>
  </div>
</template>

<script>
import fater from "./components/fater";

export default {
  name: 'App',
  components: {
    fater
  },
 data(){
    return{
    }
 },
  methods:{
    faterMethod(){
      console.log('父组件中的方法');
    }
  }
}
</script>

</style>

子组件：

<template>
    <div>
        子组件

    </div>

            getFaterMeth(){

# vue的路由

使用和react的其实大同小异

  1. 安装>npm install vue-router -s
  2. 引用import VueRouter from 'vue-router' Vue.use(VueRouter)
  3. main.js中配置：
const router = new VueRouter({
  routes:[
    {path:'/xxx',component:xxx} 
  ]
})
new Vue({
  render: h => h(App),
  router
}).$mount('#app')
  1. 使用router-link to='/xxx/'占位router-view
