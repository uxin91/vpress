# 原生js实现拖动色块变色

想实现一个这样的效果。用拖动的方式，给目标盒子上色，而且刷新后依然保存上的色，并设置清除按钮。

  • 开始布局
<div class='ul'>
	<div class='li'></div>
	<div class='li'></div>
	<div class='li'></div>
</div>
<div id='div'></div>
<button id='cle'>清除缓存</button>
  • 设置样式
		.ul{
			display: flex;
			justify-content: space-between;
			margin-bottom: 300px;
		}
		.li{
			width: 100px;
			height: 100px;
			text-align: center;
			line-height: 100px;
			list-style: none;
		}
		.li:nth-child(1){
			background-color: red;
		}
		.li:nth-child(2){
			background-color:  yellow;
		}
		.li:nth-child(3){
			background-color: green;
		}
		#div{
			width:200px;
			height: 200px;
			border: 1px solid black;
			margin:0 auto;
		}
  • 设置让盒子允许拖动（这里设置li标签不好使，所以用的div）
<div class='li' draggable="true"></div>

window.onload = function(){
		//获取元素
	let lis = document.getElementsByClassName('li');
	let dis = document.getElementById('div');
	let cle = document.getElementById('cle');
	//判断有没有保存颜色
	if(localStorage.getItem('cols')){
		dis.style.backgroundColor=localStorage.getItem('cols');
	}
	//删除颜色
	cle.onclick = function(){
		localStorage.removeItem('cols');
	}
	
	for(let a= 0;a<lis.length;a++){
		//设置当元素拖动到盒子上方时 消除容器的默认属性
		dis.ondragover = function(e){
			e.preventDefault();
		}
		//设置拖动结束时 设置颜色
		lis[a].ondragend = function(e){
			//console.log(this);
			//获取css样式中的属性
			let cl = document.defaultView.getComputedStyle(this, null); 
			//console.log(test.backgroundColor)
			let cols = cl.backgroundColor
			//设置样式属性
			dis.style.backgroundColor = cols
			//保存结果
			localStorage.setItem('cols',cols);
		}
	}
	}
