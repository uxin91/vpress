# oracle数据库操作

数据导入导出，用户创建授权

创建表空间指定用户授权
create tablespace smarter  
logging  
datafile 'D:\app\Administrator\oradata\orcl\smarter02.dbf' 
size 50m  
autoextend on  
next 50m maxsize 30720m  
extent management local; 

create user gps_reb identified by 123456 default tablespace smarter;

grant connect,resource,dba to gps_reb;

导入:

IMP gps_reb/123456@172.15.110.87/ORCL2 file=D:\dataa\9.dmp full=y tables=(R_WEAK_OLAP)

导出:

exp gps_reb/123456@172.15.110.56/orcl file=D:\expdp\2128.dmp tables=SRC_GPS_20180121_U tables=(R_WEAK_OLAP)
